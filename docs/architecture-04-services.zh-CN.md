```
│   │                                                                     │  │
```

### 2.3 模型适配器模式

```typescript
// src/services/ai/adapters/base.ts

export interface ModelAdapter {
  // 创建请求
  createRequest(params: UnifiedRequestParams): any
  
  // 解析响应
  parseResponse(response: any): Promise<UnifiedResponse>
}

export interface UnifiedRequestParams {
  messages: MessageParam[]
  systemPrompt: string[]
  tools: Tool[]
  maxTokens: number
  stream: boolean
  temperature?: number
  reasoningEffort?: 'low' | 'medium' | 'high'
  previousResponseId?: string
  verbosity?: 'low' | 'medium' | 'high'
  stopSequences?: string[]
}

export interface UnifiedResponse {
  content: ContentBlock[]
  toolCalls?: ToolCall[]
  usage: {
    inputTokens: number
    outputTokens: number
    promptTokens?: number
    completionTokens?: number
  }
  responseId?: string
  stopReason?: string
}
```

```
┌─────────────────────────────────────────────────────────────────────────────┐
```
```
│                       MCP 服务发现流程                                      │
│                                                                             │
│   ┌─────────────────────────────────────────────────────────────────────┐  │
│   │                     配置来源                                        │  │
│   │                                                                     │  │
│   │   1. 项目配置：.kode/.mcp.json                                      │  │
│   │   2. 用户配置：~/.kode/mcp.json                                     │  │
│   │   3. 插件配置：.kode-plugin/plugin.json 中的 mcpServers            │  │
│   │                                                                     │  │
│   └─────────────────────────────────────────────────────────────────────┘  │
│                                    │                                        │
│                                    ▼                                        │
│   ┌─────────────────────────────────────────────────────────────────────┐  │
│   │                    MCP Server 配置示例                              │  │
│   │                                                                     │  │
│   │   {                                                                 │  │
│   │     "mcpServers": {                                                 │  │
│   │       "github": {                                                   │  │
│   │         "command": "mcp-server-github",                             │  │
│   │         "args": ["--token", "${GITHUB_TOKEN}"],                     │  │
│   │         "env": {                                                    │  │
│   │           "GITHUB_TOKEN": "ghp_xxx"                                 │  │
│   │         }                                                           │  │
│   │       },                                                            │  │
│   │       "database": {                                                 │  │
│   │         "url": "http://localhost:8080/mcp",                         │  │
│   │         "headers": {                                                │  │
│   │           "Authorization": "Bearer xxx"                             │  │
│   │         }                                                           │  │
│   │       }                                                             │  │
│   │     }                                                               │  │
│   │   }                                                                 │  │
│   │                                                                     │  │
│   └─────────────────────────────────────────────────────────────────────┘  │
│                                    │                                        │
│                                    ▼                                        │
│   ┌─────────────────────────────────────────────────────────────────────┐  │
│   │                    启动 MCP Server                                  │  │
│   │                                                                     │  │
│   │   if (command) {                                                    │  │
│   │     // stdio transport: 启动子进程                                  │  │
│   │     child_process.spawn(command, args, { env })                     │  │
│   │   } else if (url) {                                                 │  │
│   │     // HTTP transport: HTTP/SSE 连接                                │  │
│   │     fetch(url, { headers })                                         │  │
│   │   }                                                                 │  │
│   │                                                                     │  │
│   └─────────────────────────────────────────────────────────────────────┘  │
│                                    │                                        │
│                                    ▼                                        │
│   ┌─────────────────────────────────────────────────────────────────────┐  │
│   │                    发现工具和资源                                   │  │
│   │                                                                     │  │
│   │   1. tools/list     → 获取可用工具列表                              │  │
│   │   2. resources/list → 获取可用资源列表                              │  │
│   │   3. prompts/list   → 获取可用提示模板                              │  │
│   │                                                                     │  │
│   │   工具命名：mcp__<server_name>__<tool_name>                         │  │
│   │   例如：mcp__github__create_issue                                   │  │
│   │                                                                     │  │
│   └─────────────────────────────────────────────────────────────────────┘  │
│                                                                             │
```

### 6.2 Hook 执行流程

```
┌─────────────────────────────────────────────────────────────────────────────┐
│                        PreToolUse Hook 执行流程                             │
│                                                                             │
│   runPreToolUseHooks({ toolName, toolInput, ... })                          │
│       │                                                                     │
│       ▼                                                                     │
│   ┌─────────────────────────────────────────────────────────────────────┐  │
│   │  1. 加载匹配的 Hook                                                 │  │
│   │                                                                     │  │
│   │     • 从项目设置加载                                                │  │
│   │     • 从插件加载                                                    │  │
│   │     • 过滤匹配当前工具的 Hook (matcher 匹配)                        │  │
│   │                                                                     │  │
│   └─────────────────────────────────────────────────────────────────────┘  │
│       │                                                                     │
│       ▼                                                                     │
│   ┌─────────────────────────────────────────────────────────────────────┐  │
│   │  2. 并行执行所有 Hook                                               │  │
│   │                                                                     │  │
│   │     Hook 类型：                                                     │  │
│   │     • command: 执行外部命令，通过 stdin 传递 JSON                   │  │
│   │     • prompt: 调用 AI 模型执行逻辑                                  │  │
│   │                                                                     │  │
│   │     Hook 输入：                                                     │  │
│   │     {                                                               │  │
│   │       session_id,                                                   │  │
│   │       tool_name,                                                    │  │
│   │       tool_input,                                                   │  │
│   │       tool_use_id,                                                  │  │
│   │       permission_mode,                                              │  │
│   │       cwd,                                                          │  │
│   │     }                                                               │  │
│   │                                                                     │  │
│   └─────────────────────────────────────────────────────────────────────┘  │
│       │                                                                     │
│       ▼                                                                     │
│   ┌─────────────────────────────────────────────────────────────────────┐  │
│   │  3. 处理 Hook 输出                                                  │  │
│   │                                                                     │  │
│   │     退出码 2: 阻止工具执行                                          │  │
│   │       → 返回 { kind: 'block', message: '...' }                      │  │
│   │                                                                     │  │
│   │     退出码 0: 允许继续                                              │  │
│   │       解析 JSON 输出：                                              │  │
│   │       {                                                             │  │
│   │         systemMessage: "添加到系统提示",                            │  │
│   │         hookSpecificOutput: {                                       │  │
│   │           permissionDecision: "allow" | "deny" | "ask",             │  │
│   │           updatedInput: { ... },  // 修改工具输入                   │  │
│   │           additionalContext: "..." // 添加上下文                    │  │
│   │         }                                                           │  │
│   │       }                                                             │  │
│   │                                                                     │  │
│   │     退出码 1: 警告但继续                                            │  │
│   │       → 添加警告消息                                                │  │
│   │                                                                     │  │
│   └─────────────────────────────────────────────────────────────────────┘  │
│       │                                                                     │
│       ▼                                                                     │
│   ┌─────────────────────────────────────────────────────────────────────┐  │
│   │  4. 返回结果                                                        │  │
│   │                                                                     │  │
│   │     {                                                               │  │
│   │       kind: 'allow' | 'block',                                      │  │
│   │       warnings: string[],                                           │  │
│   │       permissionDecision: 'allow' | 'ask' | undefined,              │  │
│   │       updatedInput: { ... } | undefined,                            │  │
│   │       systemMessages: string[],                                     │  │
│   │       additionalContexts: string[],                                 │  │
│   │     }                                                               │  │
│   │                                                                     │  │
│   └─────────────────────────────────────────────────────────────────────┘  │
│                                                                             │
└─────────────────────────────────────────────────────────────────────────────┘
```
